var Constructor = function()
{
	@name
    this.damageTable = [];

    this.getName = function()
    {
        return qsTr("@name");
    };
	@custom
    this.getBaseDamage = function(unit)
    {
        return WEAPON.getDamageFromTable(unit, @obj.damageTable, "@obj");
    };
};

Constructor.prototype = WEAPON;
var @obj = new Constructor();
